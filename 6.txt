#更新源码
git clone https://github.com/hcchuan/ccc.git

(如果项目文件夹ccc已存在，则执行 cd ccc 进入项目，然后执行git pull更新代码)

#进入托管项目
cd ccc

#提交更改(add后面加空格再加点)
git add .   

#提交到本地的版本控制库里，引号里面是你对本次提交的说明信息。
git  commit -m "changes log"

#将你本地的仓库提交到你的github账号里，此时会要求你输入你的github的账号和密码。
git push -u origin master  

#回到你的github.com中的项目页，看看里面的文件是不是已经发生改变了？
